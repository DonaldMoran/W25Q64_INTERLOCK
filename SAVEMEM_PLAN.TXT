case CMD_FS_SAVEMEM:
{
    // Parse args: filename (null-terminated) + 2-byte length (little-endian)
    char *filename = (char *)arg_buf;
    
    // Find the null terminator to locate length bytes
    int filename_len = strlen(filename);
    uint16_t length = arg_buf[filename_len + 1] | (arg_buf[filename_len + 2] << 8);
    
    // Validate
    if (!fs_mounted) {
        printf("SAVEMEM: FS not mounted\n");
        bus_write_byte(STATUS_ERR);
        bus_write_byte(0); bus_write_byte(0);
        break;
    }
    
    if (length == 0 || length > 65535) {
        printf("SAVEMEM: Invalid length %d\n", length);
        bus_write_byte(STATUS_ERR);
        bus_write_byte(0); bus_write_byte(0);
        break;
    }
    
    printf("SAVEMEM: File='%s', Length=%u\n", filename, length);
    
    // Open file (create/overwrite)
    int err = lfs_file_open(&lfs, &stream_file, filename,
                           LFS_O_WRONLY | LFS_O_CREAT | LFS_O_TRUNC);
    if (err < 0) {
        printf("SAVEMEM: File open failed, err=%d\n", err);
        bus_write_byte(STATUS_ERR);
        bus_write_byte(0); bus_write_byte(0);
        break;
    }
    
    // Send OK header (ready signal)
    bus_write_byte(STATUS_OK);
    bus_write_byte(0);
    bus_write_byte(0);
    
    // Enter streaming state
    stream_bytes_remaining = length;
    current_state = STATE_SAVEMEM_STREAM;
    printf("SAVEMEM: Ready to receive %u bytes\n", length);
    break;

; ---------------------------------------------------------------------------
; do_savemem - Save memory range to file
; Format: SAVEMEM <start_hex> <end_hex> <filename>
; ---------------------------------------------------------------------------
do_savemem:
    ; First, ensure filesystem is mounted
    jsr guard_requires_mount
    bcc @proceed
    rts
    
@proceed:
    ; Parse start address (first argument)
    jsr get_first_arg
    jsr parse_hex_word
    bcs @parse_error
    sta savemem_start
    stx savemem_start+1
    
    ; Parse end address (second argument)
    jsr get_next_arg
    bcs @arg_error
    jsr parse_hex_word
    bcs @parse_error
    sta savemem_end
    stx savemem_end+1
    
    ; Get filename (third argument)
    jsr get_next_arg
    bcs @arg_error
    
    ; Copy filename to filename_buf
    ldy #0
@copy_filename:
    lda (arg_ptr), y
    sta filename_buf, y
    beq @filename_done
    iny
    cpy #31
    bne @copy_filename
    lda #0
    sta filename_buf, y
@filename_done:

    ; Compute length = end - start + 1
    sec
    lda savemem_end
    sbc savemem_start
    sta savemem_len
    lda savemem_end+1
    sbc savemem_start+1
    sta savemem_len+1
    
    ; Add 1
    clc
    lda savemem_len
    adc #1
    sta savemem_len
    lda savemem_len+1
    adc #0
    sta savemem_len+1
    
    ; Validate length (non-zero)
    lda savemem_len
    ora savemem_len+1
    bne @len_ok
    jmp @invalid_range
    
@len_ok:
    ; Pack arguments for Pico: filename (null-term) + length (2 bytes)
    ldx #0
@pack_filename:
    lda filename_buf, x
    sta ARG_BUFF, x
    beq @pack_length
    inx
    cpx #32
    bne @pack_filename
    
@pack_length:
    ; Store length after filename's null
    lda savemem_len
    sta ARG_BUFF+1, x
    lda savemem_len+1
    sta ARG_BUFF+2, x
    
    ; ARG_LEN = filename length (including null) + 2
    txa
    clc
    adc #3
    sta ARG_LEN
    
    ; Send command to Pico
    lda #CMD_FS_SAVEMEM
    sta CMD_ID
    jsr pico_send_request
    bcc @cmd_sent
    jsr timeout_error
    rts
    
@cmd_sent:
    lda LAST_STATUS
    cmp #STATUS_OK
    beq @start_stream
    jsr print_command_failed
    rts
    
@start_stream:
    ; Initialize streaming pointers
    lda savemem_start
    sta savemem_ptr
    lda savemem_start+1
    sta savemem_ptr+1
    
    lda savemem_len
    sta savemem_cnt
    lda savemem_len+1
    sta savemem_cnt+1
    
    ; Print "Saving..." message
    ldx #0
@save_msg:
    lda SAVING_MSG, x
    beq @stream_loop
    jsr OUTCH
    inx
    jmp @save_msg
    
@stream_loop:
    ; Check if done
    lda savemem_cnt
    ora savemem_cnt+1
    beq @stream_done
    
    ; Send next byte
    ldy #0
    lda (savemem_ptr), y
    jsr send_byte
    
    ; Increment pointer
    inc savemem_ptr
    bne @no_carry
    inc savemem_ptr+1
@no_carry:
    
    ; Decrement counter
    lda savemem_cnt
    bne @dec_low
    dec savemem_cnt+1
@dec_low:
    dec savemem_cnt
    
    ; Optional progress dot every 256 bytes
    lda savemem_cnt
    and #$FF
    cmp #$FF
    bne @stream_loop
    lda #'.'
    jsr OUTCH
    
    jmp @stream_loop
    
@stream_done:
    ; Read final status from Pico
    jsr read_byte
    cmp #STATUS_OK
    bne @stream_error
    
    ; Print success message
    jsr CRLF
    ldx #0
@success_msg:
    lda SAVED_MSG, x
    beq @done
    jsr OUTCH
    inx
    jmp @success_msg
    
@done:
    jsr CRLF
    rts
    
@stream_error:
    jsr print_command_failed
    rts
    
@parse_error:
    ldx #0
@parse_err_msg:

    jsr OUTCH
    inx
    jmp @parse_err_msg
@parse_err_done:
    jsr CRLF

@arg_error:

; ---------------------------------------------------------------------------
; parse_hex_word - Parse hex word from (arg_ptr)
; Returns: A/X = 16-bit value, Carry set on error
; ---------------------------------------------------------------------------
parse_hex_word:
    ldy #0
    lda #0
    sta $F0               ; Temp high byte
    sta $F1               ; Temp low byte
    
@skip_spaces:
    lda (arg_ptr), y
    cmp #' '
    bne @parse
    iny
    bne @skip_spaces
    
@parse:
    tya
    clc
    adc arg_ptr
    sta arg_ptr
    bcc @parse_loop
    inc arg_ptr+1
    
@parse_loop:
    ldy #0
    lda (arg_ptr), y
    beq @done
    cmp #' '
    beq @done
    
    jsr hex_nibble
    bcs @error
    
    ; Shift current value left 4 bits
    asl $F1
    rol $F0
    asl $F1
    rol $F0
    asl $F1
    rol $F0
    asl $F1
    rol $F0
    
    ; Add new nibble
    ora $F1
    sta $F1
    bcc @next_char
    inc $F0
    
@next_char:
    inc arg_ptr
    bne @parse_loop
    inc arg_ptr+1
    jmp @parse_loop
    
@done:
    lda $F1
    ldx $F0
    clc
    rts
    
@error:
    sec
    rts

hex_nibble:
    cmp #'0'
    bcc @not_hex
    cmp #'9'+1
    bcs @check_upper
    sec
    sbc #'0'
    clc
    rts
@check_upper:
    cmp #'A'
    bcc @not_hex
    cmp #'F'+1
    bcs @check_lower
    sec
    sbc #'A'-10
    clc
    rts
@check_lower:
    cmp #'a'
    bcc @not_hex
    cmp #'f'+1
    bcs @not_hex
    sec
    sbc #'a'-10
    clc
    rts
@not_hex:
    sec
    rts

With this design will I be able to save memory ranges larger than 255 bytes to one file, for example assume I wanted to save 512 bytes of 6502 memory to one file, does our design allow for that?
Yes, absolutely! Our design fully supports saving memory ranges larger than 255 bytes. Let me explain why and show you exactly how it works for your 512-byte example.

Why Our Design Handles >255 Bytes
The key is that we're using 16-bit length values throughout the design, not 8-bit. Here's the breakdown:

1. Length Storage (16-bit)
In both the 6502 and Pico code, length is stored as a 16-bit value:

6502 Side:

assembly
savemem_len:      .res 2   ; 2 bytes = 16-bit, supports up to 65535 bytes
savemem_cnt:      .res 2   ; Counter also 16-bit
Pico Side:

c
uint16_t stream_bytes_remaining = 0;  // 16-bit unsigned, up to 65535
uint16_t length = arg_buf[filename_len + 1] | (arg_buf[filename_len + 2] << 8);
2. Protocol Packing (16-bit)
When we pack the arguments, length uses 2 bytes:

assembly
; After filename and null:
sta ARG_BUFF+1, x    ; Length low byte
sta ARG_BUFF+2, x    ; Length high byte
3. Streaming Loop (16-bit counter)
The streaming loop properly handles 16-bit decrement:

assembly
@stream_loop:
    lda savemem_cnt
    ora savemem_cnt+1    ; Tests both high and low bytes
    beq @stream_done     ; Branch only when both are zero
    
    ; ... send byte ...
    
    ; Decrement 16-bit counter
    lda savemem_cnt
    bne @dec_low
    dec savemem_cnt+1    ; Decrement high byte when low wraps
@dec_low:
    dec savemem_cnt      ; Decrement low byte
Example: Saving 512 Bytes ($0200)
Let's trace through your 512-byte example:

Command Input:
text
SAVEMEM 1000 11FF test.bin
(start $1000, end $11FF = 512 bytes)

6502 Side Computation:
assembly
; Length = $11FF - $1000 + 1 = $200 (512 decimal)
; This is stored as:
savemem_len = $0200     ; High byte = $02, Low byte = $00
Argument Packing:
text
ARG_BUFF: "test.bin" [0] [$00] [$02]
                          │      └─ High byte ($02)
                          └──────── Low byte ($00)
Note: Little-endian format (low byte first)

Pico Side Reception:
c
uint16_t length = arg_buf[filename_len + 1] | (arg_buf[filename_len + 2] << 8);
// For "test.bin" (8 chars + null = 9 bytes):
// arg_buf[9] = $00, arg_buf[10] = $02
// length = $00 | ($02 << 8) = $0200 (512)
Streaming Process:
The Pico will receive exactly 512 bytes, with the counter properly handling the wrap from $0001 to $0000 after the last byte.

Maximum Supported Size
Our design supports up to 65,535 bytes (64KB - 1) in a single SAVEMEM command. This is because:

16-bit length = 0 to 65535

6502 address space is 64KB, so this covers any contiguous range

Pico's RAM can buffer this (your http_buf is 32KB, but we're writing directly to flash, not buffering)

Important Note About Maximum
The maximum contiguous range is actually 65,535 bytes (from $0000 to $FFFE), not $0000 to $FFFF. This is because:

$FFFF - $0000 + 1 = 65,536, which doesn't fit in 16 bits

Our counter goes from 65535 down to 0

To save the entire 64KB, you'd need to do it in two saves:

text
SAVEMEM 0000 7FFF lower32k.bin    # 32768 bytes
SAVEMEM 8000 FFFF upper32k.bin    # 32768 bytes
Code That Enables This
The critical pieces that support >255 bytes are:

16-bit variables in both 6502 and Pico code

Two-byte length in the protocol

Proper 16-bit decrement logic in the streaming loop

Little-endian packing/unpacking that works for values >255
